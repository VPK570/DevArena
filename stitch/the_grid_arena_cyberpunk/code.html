import math def solve_permutation(n, k): # Calculate k-th lexicographical sequence numbers = list(range(1, n + 1)) permutation = "" k -= 1 for i in range(n, 0, -1): idx = k // math.factorial(i - 1) permutation += str(numbers.pop(idx)) k %= math.factorial(i - 1) return permutation # Operational input prompt print(solve_permutation(3, 3))
